// server.js - Express server with Kimi API
const express = require('express');
const axios = require('axios');
const app = express();

app.use(express.json());
app.use(express.static('public'));

// Kimi API Config
const KIMI_CONFIG = {
  baseURL: 'https://api.moonshot.cn/v1',
  apiKey: 'sk-kim...', // <-- TERA API KEY YAHAN DAL
  model: 'kimi-for-coding'
};

// Chat endpoint
app.post('/chat', async (req, res) => {
  try {
    const { message } = req.body;
    
    const response = await axios.post(
      `${KIMI_CONFIG.baseURL}/chat/completions`,
      {
        model: KIMI_CONFIG.model,
        messages: [
          { role: 'system', content: 'You are a helpful coding assistant.' },
          { role: 'user', content: message }
        ],
        temperature: 0.7,
        max_tokens: 4096
      },
      {
        headers: {
          'Authorization': `Bearer ${KIMI_CONFIG.apiKey}`,
          'Content-Type': 'application/json'
        }
      }
    );
    
    res.json({
      success: true,
      response: response.data.choices[0].message.content
    });
    
  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

// Health check
app.get('/', (req, res) => {
  res.send('Kimi API Server Running!');
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});

# app.py
from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

KIMI_CONFIG = {
    "base_url": "https://api.moonshot.cn/v1",
    "api_key": "sk-kim...",  # <-- TERA API KEY YAHAN DAL
    "model": "kimi-for-coding"
}

@app.route('/chat', methods=['POST'])
def chat():
    try:
        data = request.json
        message = data.get('message', '')
        
        response = requests.post(
            f"{KIMI_CONFIG['base_url']}/chat/completions",
            headers={
                "Authorization": f"Bearer {KIMI_CONFIG['api_key']}",
                "Content-Type": "application/json"
            },
            json={
                "model": KIMI_CONFIG['model'],
                "messages": [
                    {"role": "system", "content": "You are a helpful coding assistant."},
                    {"role": "user", "content": message}
                ],
                "temperature": 0.7,
                "max_tokens": 4096
            }
        )
        
        result = response.json()
        return jsonify({
            "success": True,
            "response": result['choices'][0]['message']['content']
        })
        
    except Exception as e:
        return jsonify({
            "success": False,
            "error": str(e)
        }), 500

@app.route('/')
def home():
    return "Kimi API Server Running!"

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=3000)
